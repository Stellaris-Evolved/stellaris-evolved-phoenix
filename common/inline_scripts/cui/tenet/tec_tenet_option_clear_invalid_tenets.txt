    limit = {
        tec_has_tenet = {
            tenet = $name$
        }
        NOT = {
            tec_is_tenet_allowed = {
                tenet = $name$
            }
        }
    }
    tec_deactivate_tenet = {
        tenet = $name$
    }
}
